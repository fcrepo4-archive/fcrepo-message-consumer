018import static com.google.common.base.Preconditions.checkNotNull;
019import static com.google.common.collect.ImmutableList.builder;
020import static com.google.common.collect.Maps.transformValues;
021import static org.slf4j.LoggerFactory.getLogger;
022
023import java.io.IOException;
024import java.lang.reflect.Type;
025import java.util.Collection;
026import java.util.List;
027import java.util.Map;
028
029import org.slf4j.Logger;
030
031import com.google.common.base.Function;
032import com.google.common.collect.ImmutableList;
033import com.google.gson.Gson;
034import com.google.gson.JsonElement;
035import com.google.gson.TypeAdapter;
036import com.google.gson.reflect.TypeToken;
037import com.google.gson.stream.JsonReader;
038import com.google.gson.stream.JsonWriter;
039
040/**
041 * Deserializes JSON maps
042 *
043 * @author ajs6f
044 * @since Dec 6, 2013
045 */
046public class NamedFieldsDeserializer extends TypeAdapter<NamedFields> {
047
048    private static final Type type = new TypeToken<List<Map<String, JsonElement>>>() {}
049            .getType();
050
051    private Gson gson;
052
053    //TODO make index-time boost somehow adjustable, or something
054    public static final Long INDEX_TIME_BOOST = 1L;
055
056    private static final Logger LOGGER = getLogger(NamedFieldsDeserializer.class);
057
058    private static Function<JsonElement, Collection<String>> jsonElement2list =
059        new Function<JsonElement, Collection<String>>() {
060
061            @Override
062            public List<String> apply(final JsonElement input) {
063                final JsonElement json =
064                    checkNotNull(input, "Cannot transform null!");
065                final ImmutableList.Builder<String> b = builder();
066                for (final JsonElement value : json.getAsJsonArray()) {
067                    b.add(value.getAsString());
068                }
069                return b.build();
070            }
071        };
072
073    @Override
074    public void write(final JsonWriter out, final NamedFields value)
075        throws IOException {
076        throw new UnsupportedOperationException();
077    }
078
079    @Override
080    public NamedFields read(final JsonReader in)
081        throws IOException {
082        try {
083            final List<Map<String, JsonElement>> fields = gson.fromJson(in, type);
084            return new NamedFields(transformValues(fields.get(0), jsonElement2list));
085        } catch (final Exception e) {
086            LOGGER.error("Failed to parse JSON to List<Map<String, Collection<String>>>!", e);
087            throw e;
088        }
089
090    }
091
092
093    /**
094     * @param gson the Gson engine to set
095     * @return this object for continued use
096     */
097    public NamedFieldsDeserializer setGson(final Gson gson) {
098        this.gson = gson;
099        return this;
100    }
101
102}
