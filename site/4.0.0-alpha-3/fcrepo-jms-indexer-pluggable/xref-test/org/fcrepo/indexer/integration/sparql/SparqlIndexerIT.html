41  @RunWith(SpringJUnit4ClassRunner.class)
42  @ContextConfiguration({"/spring-test/test-container.xml"})
43  public class SparqlIndexerIT {
44  
45      private static final String uri = "http://example.com/sparqlIndexerTestURI";
46  
47  
48      private static final Logger LOGGER = getLogger(SparqlIndexerIT.class);
49  
50      @Inject
51      private SparqlIndexer sparqlIndexer;
52  
53      private static final String fooRDF =
54          "@prefix fcrepo: <http://fcrepo.org/repository#> .\n" +
55          "@prefix fedora: <http://fcrepo.org/repository/rest-api#> .\n" +
56          "<" + uri + ">\n" +
57          " fcrepo:hasChild <" + uri + "/barDS> ;\n" +
58          " fcrepo:hasParent <" + uri + "> ;\n" +
59          " fcrepo:uuid \"feb99ff2-455e-4e16-93a0-c0ae8d21b9ae\" .";
60  
61      @Test
62      public void indexerTest() throws Exception {
63          // add triples
64          sparqlIndexer.update(uri, createDefaultModel().read(
65                  new StringReader(fooRDF), "", "N3"));
66  
67          waitForTriples(3);
68  
69          // triples should be present in the triplestore
70          assertEquals("Triples should be present!", 3, sparqlIndexer
71                  .countTriples(uri));
72  
73          // remove object
74          sparqlIndexer.remove(uri);
75  
76          waitForTriples(0);
77  
78          // triples should not be present in the triplestore
79          assertTrue("Triples should not be present!", sparqlIndexer
80                  .countTriples(uri) == 0);
81      }
82  
83      private void waitForTriples(final int expectTriples) throws InterruptedException {
84          long elapsed = 0;
85          final long restingWait = 500;
86          final long maxWait = 15000; // 15 seconds
87  
88          int count = sparqlIndexer.countTriples(uri);
89          while ((count < expectTriples) && (elapsed < maxWait)) {
90              LOGGER.debug("Discovered {} triples, waiting for {}...", count, expectTriples);
91              sleep(restingWait);
92              count = sparqlIndexer.countTriples(uri);
93  
94              elapsed += restingWait;
95          }
96      }

98  }
