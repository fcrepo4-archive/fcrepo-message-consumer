053/**
054 * Indexes triples from Fedora into a triplestore using SPARQL Update.
055 *
056 * @author Esm√© Cowles
057 * @author ajs6f
058 * @since Aug 19, 2013
059**/
060public class SparqlIndexer extends AsynchIndexer<Model, Void> {
061
062    private String queryBase;
063    private String updateBase;
064    private boolean formUpdates = false;
065
066    private static final Logger LOGGER = getLogger(SparqlIndexer.class);
067
068    /**
069     * Number of threads to use for operating against the triplestore.
070     */
071    private static final Integer THREAD_POOL_SIZE = 5;
072
073    private ListeningExecutorService executorService =
074        listeningDecorator(newFixedThreadPool(THREAD_POOL_SIZE));
075
076    /**
077     * Remove any current triples about the Fedora object and replace them with
078     * the provided content.
079     * @content RDF in N3 format.
080    **/
081    @Override
082    public Callable<Void> updateSynch(final URI pid, final Model model) {
083        LOGGER.debug("Received update for: {}", pid);
084        removeSynch(pid, true);
085        // build a list of triples
086        final StmtIterator triples = model.listStatements();
087        final QuadDataAcc add = new QuadDataAcc();
088        while ( triples.hasNext() ) {
089            add.addTriple( triples.nextStatement().asTriple() );
090        }
091
092        // send update to server
093        LOGGER.debug("Sending update request for pid: {}", pid);
094        return exec(new UpdateRequest(new UpdateDataInsert(add)));
095    }
096
097    /**
098     * Perform a DESCRIBE query for triples about the Fedora object and remove
099     * all triples with subjects starting with the same subject.
100    **/
101    @Override
102    public Callable<Void> removeSynch(final URI subject) {
103        return removeSynch(subject, false);
104    }
105
106    @VisibleForTesting
107    protected Callable<Void> removeSynch(final URI subject, final boolean blocking) {
108
109        LOGGER.debug("Received remove for: {}", subject);
110        // find triples/quads to delete
111        final String describeQuery = "DESCRIBE <" + subject.toString() + ">";
112        final QueryEngineHTTP qexec = buildQueryEngineHTTP(describeQuery);
113        final Iterator<Triple> results = qexec.execDescribeTriples();
114
115        // build list of triples to delete
116        final Set<String> uris = new HashSet<>();
117        while ( results.hasNext() ) {
118            final Triple triple = results.next();
119
120            // add subject uri, if it is part of this object
121            if ( triple.getSubject().isURI() ) {
122                final String uri = ((Node_URI)triple.getSubject()).getURI();
123                if ( matches(subject, uri) ) {
124                    uris.add(uri);
125                }
126            }
127
128            // add object uri, if it is part of this object
129            if ( triple.getObject().isURI() ) {
130                final String uri = ((Node_URI)triple.getObject()).getURI();
131                if ( matches(subject, uri) ) {
132                    uris.add(uri);
133                }
134            }
135        }
136        qexec.close();
137
138        // build update commands
139        final UpdateRequest del = buildUpdateRequest();
140        for (final String uri : uris) {
141            final String cmd = "DELETE WHERE { <" + uri + "> ?p ?o }";
142            LOGGER.debug("Executing: {}", cmd);
143            del.add(cmd);
144        }
145
146        // send updates
147        return exec(del, blocking);
148    }
149
150    /**
151     * Determine whether arg candidate is a sub-URI of arg resource, defined as candidate-URI starting
152     * with resource-URI, plus an option suffix starting with a hash (#) or slash (/)
153     * suffix.
154    **/
155    private boolean matches( final URI resource, final String candidate) {
156        // All triples that will match this logic are ones that:
157        // - have a candidate subject or object that equals the target resource of removal, or
158        // - have a candidate subject or object that is prefixed with the resource of removal
159        //    (therefore catching all children).
160        return resource.toString().equals(candidate) || candidate.startsWith(resource.toString() + "/")
161            || candidate.startsWith(resource.toString() + "#");
162    }
163
164    private Callable<Void> exec(final UpdateRequest update) {
165        return exec(update, false);
166    }
167
168    private Callable<Void> exec(final UpdateRequest update, final boolean blocking) {
169        if (update.getOperations().isEmpty()) {
170            LOGGER.debug("Received empty update/remove operation.");
171            return new Callable<Void>() {
172
173                @Override
174                public Void call() {
175                    return null;
176                }
177            };
178        }
179
180        final Callable<Void> callable = new Callable<Void>() {
181
182            @Override
183            public Void call() {
184
185                if (formUpdates) {
186                    // form updates
187                    final UpdateProcessor proc =
188                        createRemoteForm(update, updateBase);
189                    proc.execute();
190                } else {
191                    // normal SPARQL updates
192                    final UpdateProcessRemote proc =
193                        new UpdateProcessRemote(update, updateBase,
194                                emptyContext);
195                    try {
196                        proc.execute();
197                    } catch (final Exception e) {
198                        LOGGER.error(
199                                "Error executing Sparql update/remove!", e);
200                    }
201                }
202                return null;
203            }
204        };
205
206        if (blocking) {
207            try {
208                callable.call();
209            } catch (Exception e) {
210                LOGGER.error("Error calling Sparql update/remove!, {}", e.getMessage());
211            }
212
213        } else {
214            final ListenableFutureTask<Void> task =
215                ListenableFutureTask.create(callable);
216            task.addListener(new Runnable() {
217
218                @Override
219                public void run() {
220                    LOGGER.debug("Completed Sparql update/removal.");
221                    if (LOGGER.isTraceEnabled()) {
222                        try (
223                            final OutputStream buffer = new ByteArrayOutputStream()) {
224                            final IndentedWriter out = new IndentedWriter(buffer);
225                            update.output(out);
226                            LOGGER.trace("Executed update/remove operation:\n{}",
227                                    buffer.toString());
228                            out.close();
229                        } catch (final IOException e) {
230                            LOGGER.error(
231                                    "Couldn't retrieve execution of update/remove operation!",
232                                    e);
233                        }
234                    }
235                }
236            }, executorService);
237            executorService.submit(task);
238        }
239
240        return callable;
241    }
242
243    @Override
244    public IndexerType getIndexerType() {
245        return RDF;
246    }
